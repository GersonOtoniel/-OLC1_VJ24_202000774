<INIT> ::= PROG instrucciones ENDPROG
          |

<instrucciones> ::= <instrucciones> <instruccion>
                  | <instruccion>

<instruccion> ::= <ARRAYS>
                | <ASIGNACION>
                | <IMPRIMIR>
                | <IMPRIMIRARR>
                | <GRAFICAS>

<ARRAYS> ::= ARR DP TIPO DP DP DP ARROBAID ASIGN LISTA END PYC

<ASIGNACION> ::= VAR DP TIPO DP DP ID ASIGN EXPRESION END PYC

<IMPRIMIR> ::= CONSOLE DP DP PRINT IGUAL EXIMPRIMIR END PYC

<IMPRIMIRARR> ::= CONSOLE DP DP PRINT IGUAL EXIMPRIMIRARR END PYC

<GRAFICAS> ::= GBAR DP DP GRAPHBARR END PYC
             | GPIE DP DP GRAPHPIE END PYC
             | GLINE DP DP GRAPHHISTO END PYC

<GRAPHPIE> ::= GPIE PARENABRE LABEL DP DP CHAR IGUAL <LISTA> END PYC
               VALUES DP DP DOUBLE IGUAL <LISTA> END PYC
               TITLE DP DP CHAR IGUAL EXPRESION END PYC
               EXEC GPIE END PYC PARENCIERRA
               END PYC

<GRAPHBARR> ::= GBAR  PARENABRE TITLE DP DP CHAR IGUAL EXPRESION END PYC
                EJEX DP DP CHAR IGUAL <LISTA> END PYC
                EJEY DP DP DOUBLE IGUAL <LISTA> END PYC
                TITLEX DP DP CHAR IGUAL EXPRESION END PYC
                TITLEY DP DP CHAR IGUAL EXPRESION END PYC
                EXEC GBAR END PYC PARENCIERRA END PYC

<LISTCOLUMN> ::= <LISTA>
                | <LISTAEXPRESION>

<LISTA> ::= CORCHABRE <LISTAEXPRESION> CORCHCIERRA

<LISTAEXPRESION> ::= <LISTAEXPRESION> COMA EXPRESION
                    | EXPRESION

<TIPO> ::= CHAR
         | DOUBLE


<EXIMPRIMIR> ::= <EXIMPRIMIR> COMA EXPRESION
              | EXPRESION
         

<EXPRESION> ::= STRING
              | ARROBAID
              | OPERA
              | ESTADISTICOS
     

<ESTADISTICOS> ::= MEDIA PARENABRE LISTA PARENCIERRA
                | MEDIANA PARENABRE LISTA PARENCIERRA
                | MODA PARENABRE LISTA PARENCIERRA
                | VARIANZA PARENABRE LISTA PARENCIERRA
		| MIN PARENABRE <LISTA> PARENCIERRA
		| MAX PARENABRE <LISTA> PARENCIERRA

<OPERA> ::= OPERA SUMA
          | SUMA PARENABRE OPERA COMA OPERA PARENCIERRA
          | RESTA PARENABRE OPERA COMA OPERA PARENCIERRA      
          | MULTIPLICACION PARENABRE OPERA COMA OPERA PARENCIERRA
	  | DIVISION PARENABRE <OPERA>:num1 COMA <OPERA>:num2 PARENCIERRA  
          | MODULO PARENABRE <OPERA> COMA <OPERA> PARENCIERRA
          | DECIMAL
          | ID
